# Menghitung-Luas-Persegi-Panjang

#include <iostream>
using namespace std;
int main ()
{
    int p,l,luas;
    cout<< "<==== Menghitung Luas Persegi Panjang ====>\n";
    cout<< "Input Panjang :";
    cin>>p;
    cout<< "Input Lebar :";
    cin>>l;
    luas=p*l;
    cout<< "Luas Persegi Panjang :"<<luas;
    return 0;
}
